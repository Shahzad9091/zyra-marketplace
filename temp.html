async function seedNadraData() { const db = firebase.firestore(); const batch = db.batch(); // We use a batch for faster writing const names = ["Ahmed Ali", "Sara Khan", "Bilal Sheikh", "Zainab Bibi", "Hamza Malik", "Dua Zehra", "Usman Ghani", "Ayesha Noor"]; console.log("Starting to seed 50 records..."); for (let i = 0; i < 50; i++) { // Generate a unique 13-digit dummy CNIC const cnic = "42101" + Math.floor(10000000 + Math.random() * 90000000); const randomName = names[Math.floor(Math.random() * names.length)] + " " + (i + 1); const docRef = db.collection("nadra_records").doc(cnic); batch.set(docRef, { name: randomName }); } try { await batch.commit(); alert("Success! 50 test CNICs added to your database."); console.log("Seeding complete."); } catch (error) { console.error("Error seeding data: ", error); } }
